Querys

Pull a number of stories

query get30Stories {
  deanzastorieslimit(first: 30) {
    header
    content
    id
  }
}

Pull all stories

query allDeAnzaStories {
  deanzastories {
    header
    content
    id
  }
}

Get story count

query getStoryCount(school:"deanza")
{
  count 
  lastupdated
}
